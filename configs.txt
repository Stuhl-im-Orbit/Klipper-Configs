# -- Print lifecycle -----------------------------------------------------------

[gcode_macro PRINT_START]
description: Execute sequence -- Main startup procedure
gcode:
    {% set vars        = printer["gcode_macro _MY_VARS"] %}
    {% set target_bed  = params.BED_TEMP|default(60.0)|float %}
    {% set target_ext  = params.EXTRUDER_TEMP|default(200.0)|float %}
    G21 G90 M83
    _RESET_STATE
    ; Start bed non-blocking, then block on hotend preheat -- both heat in parallel
    SET_HEATER_TEMPERATURE HEATER=heater_bed TARGET={target_bed}
    _NOTIFY MSG="Process: Waiting for nozzle to {vars.cfg_preheat_ext}C..."
    M109 S{vars.cfg_preheat_ext}
    _CONDITIONAL_HOME
    CENTER
    G90
    G0 Z{vars.cfg_z_hop_heat} F{vars.calc_travel_speed_z}
    _NOTIFY MSG="Process: Waiting for bed {target_bed}C..."
    M190 S{target_bed}
    BED_MESH_PROFILE LOAD=default ; Load active bed mesh
    ; Move to purge position before final heat-up -- prevents ooze onto bed center
    G0 Z{vars.cfg_z_hop_heat} F{vars.calc_travel_speed_z}
    G0 X{vars.cfg_purge_x} Y{vars.cfg_purge_y} F{vars.calc_travel_speed_xy}
    _NOTIFY MSG="Process: Heating nozzle to {target_ext}C..."
    M109 S{target_ext}
    PURGE  ; Execute priming
    _NOTIFY MSG="Process: Print started."

[gcode_macro PRINT_END]
description: Execute sequence -- Finalize print job
gcode:
    M400
    G10
    TURN_OFF_HEATERS
    M107
    _TOOLHEAD_PARK_PAUSE_CANCEL
    M84
    _RESET_STATE
    _NOTIFY MSG="Process: Print finished."

# -- Extruder & filament -------------------------------------------------------

[gcode_macro PURGE]
description: Extrude plastic -- Prime nozzle line
gcode:
    {% set vars = printer["gcode_macro _MY_VARS"] %}
    {% if printer.extruder.can_extrude %}
        _NOTIFY MSG="Process: Priming nozzle..."
        SAVE_GCODE_STATE NAME=STATE_PURGE
        G90
        M83
        G0 Z{vars.cfg_z_hop_heat} F{vars.calc_travel_speed_z}
        G0 X{vars.cfg_purge_x} Y{vars.cfg_purge_y} F{vars.calc_travel_speed_xy}
        G0 Z{vars.cfg_purge_z} F{vars.calc_travel_speed_z}
        G1 E5 F{vars.cfg_speed_purge}
        G1 X{vars.cfg_purge_x + 20.0} E20 F{vars.cfg_speed_purge}
        G10
        G0 Y{vars.cfg_purge_y + 10.0} Z{vars.cfg_purge_z + 0.8} F{vars.calc_travel_speed_xy}
        M400
        RESTORE_GCODE_STATE NAME=STATE_PURGE
    {% else %}
        _NOTIFY MSG="System Error: Extruder cold. Cannot purge."
    {% endif %}

[gcode_macro LOAD_FILAMENT]
description: Load filament -- Temperature priority managed
gcode:
    {% set vars        = printer["gcode_macro _MY_VARS"] %}
    {% set min_temp    = printer.configfile.settings.extruder.min_extrude_temp|float %}
    {% set param_temp  = params.TEMP|default(0)|float %}
    {% set cur_target  = printer.extruder.target|float %}
    {% set saved_temp  = printer["gcode_macro _PRINT_STATE"].saved_ext_temp|float %}
    {% set ns = namespace(target=vars.cfg_fallback_temp_ext) %}
    {% if cur_target >= min_temp %}{% set ns.target = cur_target  %}{% endif %}
    {% if saved_temp >= min_temp %}{% set ns.target = saved_temp  %}{% endif %}
    {% if param_temp >= min_temp %}{% set ns.target = param_temp  %}{% endif %}
    {% if ns.target >= min_temp %}
        _NOTIFY MSG="Filament: Heating to {ns.target}C..."
        SAVE_GCODE_STATE NAME=STATE_LOAD_FILAMENT
        M109 S{ns.target}
        M83
        _NOTIFY MSG="Filament: Loading {vars.cfg_fil_load_dist}mm..."
        G1 E{vars.cfg_fil_load_dist} F600
        G4 P1000
        _NOTIFY MSG="Filament: Purging {vars.cfg_fil_purge_dist}mm..."
        G1 E{vars.cfg_fil_purge_dist} F100
        G10
        M400
        RESTORE_GCODE_STATE NAME=STATE_LOAD_FILAMENT
        ; Clear saved state only after successful load -- prevents interference with manual swaps
        SET_GCODE_VARIABLE MACRO=_PRINT_STATE VARIABLE=saved_ext_temp VALUE=0.0
        _NOTIFY MSG="Filament: Load complete. Hotend at {ns.target}C."
    {% else %}
        _NOTIFY MSG="System Error: Target {ns.target}C below min_extrude_temp. Aborting."
    {% endif %}

[gcode_macro UNLOAD_FILAMENT]
description: Unload filament -- Temperature priority managed
gcode:
    {% set vars       = printer["gcode_macro _MY_VARS"] %}
    {% set min_temp   = printer.configfile.settings.extruder.min_extrude_temp|float %}
    {% set param_temp = params.TEMP|default(0)|float %}
    {% set cur_target = printer.extruder.target|float %}
    {% set saved_temp = printer["gcode_macro _PRINT_STATE"].saved_ext_temp|float %}
    {% set ns = namespace(target=vars.cfg_fallback_temp_ext) %}
    {% if cur_target >= min_temp %}{% set ns.target = cur_target  %}{% endif %}
    {% if saved_temp >= min_temp %}{% set ns.target = saved_temp  %}{% endif %}
    {% if param_temp >= min_temp %}{% set ns.target = param_temp  %}{% endif %}
    {% if ns.target >= min_temp %}
        _NOTIFY MSG="Filament: Heating to {ns.target}C..."
        SAVE_GCODE_STATE NAME=STATE_UNLOAD_FILAMENT
        M109 S{ns.target}
        M83
        _NOTIFY MSG="Filament: Shaping tip..."
        ; Tip shaping -- push then rapidly cycle to break strings before unload
        G1 E10.0 F300
        G1 E-15.0 F3000
        {% for _ in range(4) %}
            G1 E2.0 F3000
            G1 E-2.0 F3000
        {% endfor %}
        _NOTIFY MSG="Filament: Unloading {vars.cfg_fil_unload_dist}mm..."
        G1 E-{vars.cfg_fil_unload_dist} F300
        M400
        RESTORE_GCODE_STATE NAME=STATE_UNLOAD_FILAMENT
        _NOTIFY MSG="Filament: Unload complete."
    {% else %}
        _NOTIFY MSG="System Error: Target {ns.target}C below min_extrude_temp. Aborting."
    {% endif %}

[gcode_macro PRIME_BLOB]
description: Extrude plastic -- Purge nozzle via prime blob
gcode:
    {% set vars       = printer["gcode_macro _MY_VARS"] %}
    {% set vol_mult   = 1.0 / ((0.4 / printer.configfile.settings.extruder.nozzle_diameter|float) ** 2) %}
    {% set e_blob     = (10.0 * vol_mult)|round(3) %}
    {% set skip_travel = params.SKIP_TRAVEL|default(0)|int %}
    {% if printer.extruder.can_extrude %}
        _NOTIFY MSG="Process: Priming nozzle..."
        _SET_LED_STATE STATE=cleaning
        _TOGGLE_FILAMENT_SENSORS ENABLE=0
        SAVE_GCODE_STATE NAME=STATE_PRIME_BLOB
        G90
        M83
        {% if not skip_travel %}
            G0 Z{vars.cfg_z_hop_home} F{vars.calc_travel_speed_z}
            G0 X{vars.cfg_purge_x} Y{vars.cfg_purge_y} F{vars.calc_travel_speed_xy}
        {% endif %}
        G0 Z0.5 F{vars.calc_travel_speed_z}
        ; Split into 2x to stay within max_extrude_cross_section; F300 presses blob into bed
        G1 E{e_blob} F300
        G1 E{e_blob} F300
        M106 S102
        G1 Z5.0 E{(5.0 * vol_mult)|round(3)} F100  ; lift + attach string to blob top
        {% set stretch_y = [vars.cfg_purge_y + 15.0, printer.toolhead.axis_maximum.y|float]|min %}
        G1 Y{stretch_y} E{(1.0 * vol_mult)|round(3)} F200  ; anchor string to bed
        ; Stepped Y+Z wipe -- Z increments prevent TUNING_TOWER abort
        {% for i in range(4) %}
            {% set loop_y = [vars.cfg_purge_y + 22.0 + (i * 7.0), printer.toolhead.axis_maximum.y|float]|min %}
            {% set loop_z = (3.8 - (i * 1.2))|round(1) %}
            G1 Y{loop_y} Z{loop_z} E{(0.5 * vol_mult)|round(3)} F200
        {% endfor %}
        M107
        {% set tail_y = [vars.cfg_purge_y + 50.0, printer.toolhead.axis_maximum.y|float]|min %}
        G1 Y{tail_y} Z0.2 E{(0.6 * vol_mult)|round(3)} F200  ; tack tail to bed
        {% set snap_x = [vars.cfg_purge_x + 50.0, printer.toolhead.axis_maximum.x|float]|min %}
        G0 X{snap_x} F{vars.calc_travel_speed_xy}  ; fast X snap -- breaks string cleanly
        M400
        RESTORE_GCODE_STATE NAME=STATE_PRIME_BLOB
        _TOGGLE_FILAMENT_SENSORS ENABLE=1
        _SET_LED_STATE STATE=ready
    {% else %}
        _NOTIFY MSG="System Error: Extruder cold. Cannot prime."
    {% endif %}

[gcode_macro PRINT_START]
description: Execute sequence -- Main startup procedure
gcode:
    {% set vars        = printer["gcode_macro _MY_VARS"] %}
    {% set target_bed  = params.BED_TEMP|default(60.0)|float %}
    {% set target_ext  = params.EXTRUDER_TEMP|default(200.0)|float %}
    G21 G90 M83
    _RESET_STATE
    ; Start bed non-blocking, then block on hotend preheat -- both heat in parallel
    SET_HEATER_TEMPERATURE HEATER=heater_bed TARGET={target_bed}
    _NOTIFY MSG="Process: Waiting for nozzle to {vars.cfg_preheat_ext}C..."
    M109 S{vars.cfg_preheat_ext}

    G90

    M190 S{target_bed}
    G28 Z

    G0 Z{vars.cfg_z_hop_heat} F{vars.calc_travel_speed_z}

    _NOTIFY MSG="Process: Heating nozzle to {target_ext}C..."
    M109 S{target_ext}

    _NOTIFY MSG="Process: Print started."

[gcode_macro PRIME_BLOB]
description: Extrude plastic -- Purge nozzle via prime blob
gcode:
    {% set vars       = printer["gcode_macro _MY_VARS"] %}
    {% set vol_mult   = 1.0 / ((0.4 / printer.configfile.settings.extruder.nozzle_diameter|float) ** 2) %}
    {% set e_blob     = (10.0 * vol_mult)|round(3) %}
    {% set skip_travel = params.SKIP_TRAVEL|default(0)|int %}
    {% if printer.extruder.can_extrude %}
        _NOTIFY MSG="Process: Priming nozzle..."
        SAVE_GCODE_STATE NAME=STATE_PRIME_BLOB
        G90
        M83
        {% if not skip_travel %}
            G0 Z{vars.cfg_z_hop_home} F{vars.calc_travel_speed_z}
            G0 X{vars.cfg_purge_x} Y{vars.cfg_purge_y} F{vars.calc_travel_speed_xy}
        {% endif %}
        G0 Z0.5 F{vars.calc_travel_speed_z}
        ; Split into 2x to stay within max_extrude_cross_section; F300 presses blob into bed
        G1 E{e_blob} F300
        G1 E{e_blob} F300
        M106 S102
        G1 Z5.0 E{(5.0 * vol_mult)|round(3)} F100  ; lift + attach string to blob top
        {% set stretch_y = [vars.cfg_purge_y + 15.0, printer.toolhead.axis_maximum.y|float]|min %}
        G1 Y{stretch_y} E{(1.0 * vol_mult)|round(3)} F200  ; anchor string to bed
        ; Stepped Y+Z wipe -- Z increments prevent TUNING_TOWER abort
        {% for i in range(4) %}
            {% set loop_y = [vars.cfg_purge_y + 22.0 + (i * 7.0), printer.toolhead.axis_maximum.y|float]|min %}
            {% set loop_z = (3.8 - (i * 1.2))|round(1) %}
            G1 Y{loop_y} Z{loop_z} E{(0.5 * vol_mult)|round(3)} F200
        {% endfor %}
        M107
        {% set tail_y = [vars.cfg_purge_y + 50.0, printer.toolhead.axis_maximum.y|float]|min %}
        G1 Y{tail_y} Z0.2 E{(0.6 * vol_mult)|round(3)} F200  ; tack tail to bed
        {% set snap_x = [vars.cfg_purge_x + 50.0, printer.toolhead.axis_maximum.x|float]|min %}
        G0 X{snap_x} F{vars.calc_travel_speed_xy}  ; fast X snap -- breaks string cleanly
        M400
        RESTORE_GCODE_STATE NAME=STATE_PRIME_BLOB
    {% else %}
        _NOTIFY MSG="System Error: Extruder cold. Cannot prime."
    {% endif %}

[gcode_macro PRINT_START]

    {% set vars        = printer["gcode_macro _MY_VARS"] %}

    {% set target_ext  = params.EXTRUDER_TEMP|default(200.0)|float %}
    {% set ns          = namespace(high_temp = target_bed >= vars.cfg_bed_high_temp) %}
    {% set target_cha  = params.CHAMBER|default(45.0 if ns.high_temp else 15.0)|float %}

    _RESET_STATE
    UPDATE_DELAYED_GCODE ID=_STOP_NEVERMORE DURATION=0
    SET_TEMPERATURE_FAN_TARGET temperature_fan=exhaust_fan target={target_cha}
    {% if ns.high_temp %}
        _NOTIFY MSG="Process: Enabling Nevermore VOC filter"
        SET_FAN_SPEED FAN=nevermore SPEED=1.0
    {% endif %}
    ; Start bed non-blocking, then block on hotend preheat -- both heat in parallel
    SET_HEATER_TEMPERATURE HEATER=heater_bed TARGET={target_bed}
    _NOTIFY MSG="Process: Waiting for nozzle to {vars.cfg_preheat_ext}C..."
    M109 S{vars.cfg_preheat_ext}
    _CONDITIONAL_HOME
    CENTER
    G90
    G0 Z{vars.cfg_z_hop_heat} F{vars.calc_travel_speed_z}
    _NOTIFY MSG="Process: Waiting for bed {target_bed}C..."
    M190 S{target_bed}
    _NOTIFY MSG="Process: Waiting for chamber {target_cha}C..."
    TEMPERATURE_WAIT SENSOR="temperature_fan exhaust_fan" MINIMUM={target_cha}
    QUAD_GANTRY_LEVEL
    G28 Z
    BED_MESH_CALIBRATE ADAPTIVE=1
    ; Move to purge corner before final heat-up -- prevents ooze onto bed center
    G0 Z{vars.cfg_z_hop_heat} F{vars.calc_travel_speed_z}
    G0 X{vars.cfg_purge_x} Y{vars.cfg_purge_y} F{vars.calc_travel_speed_xy}
    _NOTIFY MSG="Process: Heating nozzle to {target_ext}C..."
    M109 S{target_ext}
    PRIME_BLOB SKIP_TRAVEL=1  ; already at purge position
    _TOGGLE_FILAMENT_SENSORS ENABLE=0  ; slicer re-enables at layer 2
    _NOTIFY MSG="Process: Print started."

# -- Print lifecycle -----------------------------------------------------------

[gcode_macro PRINT_START]
description: Execute sequence -- Main startup procedure
gcode:
    {% set vars        = printer["gcode_macro _MY_VARS"] %}
    {% set target_bed  = params.BED_TEMP|default(60.0)|float %}
    {% set target_ext  = params.EXTRUDER_TEMP|default(200.0)|float %}
    {% set ns          = namespace(high_temp = target_bed >= vars.cfg_bed_high_temp) %}
    {% set target_cha  = params.CHAMBER|default(45.0 if ns.high_temp else 15.0)|float %}
    G21 G90 M83
    _RESET_STATE
    UPDATE_DELAYED_GCODE ID=_STOP_NEVERMORE DURATION=0
    SET_TEMPERATURE_FAN_TARGET temperature_fan=exhaust_fan target={target_cha}
    {% if ns.high_temp %}
        _NOTIFY MSG="Process: Enabling Nevermore VOC filter"
        SET_FAN_SPEED FAN=nevermore SPEED=1.0
    {% endif %}
    ; Start bed non-blocking, then block on hotend preheat -- both heat in parallel
    SET_HEATER_TEMPERATURE HEATER=heater_bed TARGET={target_bed}
    _NOTIFY MSG="Process: Waiting for nozzle to {vars.cfg_preheat_ext}C..."
    M109 S{vars.cfg_preheat_ext}
    _CONDITIONAL_HOME
    CENTER
    G90
    G0 Z{vars.cfg_z_hop_heat} F{vars.calc_travel_speed_z}
    _NOTIFY MSG="Process: Waiting for bed {target_bed}C..."
    M190 S{target_bed}
    _NOTIFY MSG="Process: Waiting for chamber {target_cha}C..."
    TEMPERATURE_WAIT SENSOR="temperature_fan exhaust_fan" MINIMUM={target_cha}
    QUAD_GANTRY_LEVEL
    G28 Z
    BED_MESH_CALIBRATE ADAPTIVE=1
    ; Move to purge corner before final heat-up -- prevents ooze onto bed center
    G0 Z{vars.cfg_z_hop_heat} F{vars.calc_travel_speed_z}
    G0 X{vars.cfg_purge_x} Y{vars.cfg_purge_y} F{vars.calc_travel_speed_xy}
    _NOTIFY MSG="Process: Heating nozzle to {target_ext}C..."
    M109 S{target_ext}
    PRIME_BLOB SKIP_TRAVEL=1  ; already at purge position
    _TOGGLE_FILAMENT_SENSORS ENABLE=0  ; slicer re-enables at layer 2
    _NOTIFY MSG="Process: Print started."

[gcode_macro PRINT_END]
description: Execute sequence -- Finalize print job
gcode:
    M400
    G10
    TURN_OFF_HEATERS
    M107
    _TOOLHEAD_PARK_PAUSE_CANCEL
    M84
    _TOGGLE_FILAMENT_SENSORS ENABLE=1
    _RESET_STATE
    _NEVERMORE_COOLDOWN MSG="Process: Print finished."

[gcode_macro _CONDITIONAL_HOME]
description: Home axes -- G28 only if axes are not already homed
gcode:
    {% if "xyz" not in printer.toolhead.homed_axes|lower %}
        _NOTIFY MSG="System: Axes not homed. Executing G28..."
        G28
    {% else %}
        _NOTIFY MSG="System: Axes already homed. Skipping G28."
    {% endif %}

[gcode_macro _HOME_AXIS]
description: Home axis -- Sensorless homing helper reduces current before tap
gcode:
    {% set axis        = params.AXIS|string|lower %}
    {% set vars        = printer["gcode_macro _MY_VARS"] %}
    {% set cur_x       = printer['tmc2209 stepper_x'].run_current|float %}
    {% set cur_y       = printer['tmc2209 stepper_y'].run_current|float %}
    {% set max_vel     = printer.configfile.settings.printer.max_velocity|float %}
    {% set max_accel   = printer.configfile.settings.printer.max_accel|float %}
    _NOTIFY MSG="System: Homing {axis|upper}..."
    SAVE_GCODE_STATE NAME=STATE_HOME_AXIS
    M400
    ; CoreXY: both motors drive every move -- reduce both regardless of target axis
    SET_TMC_CURRENT STEPPER=stepper_x CURRENT={vars.cfg_homing_current}
    SET_TMC_CURRENT STEPPER=stepper_y CURRENT={vars.cfg_homing_current}
    ; Low accel prevents false StallGuard triggers from high-jerk moves
    SET_VELOCITY_LIMIT VELOCITY=50 ACCEL=1000
    G4 P1000
    G28 {axis|upper}
    G91
    G0 {axis|upper}-10 F1200
    G90
    SET_VELOCITY_LIMIT VELOCITY={max_vel} ACCEL={max_accel}
    SET_TMC_CURRENT STEPPER=stepper_x CURRENT={cur_x}
    SET_TMC_CURRENT STEPPER=stepper_y CURRENT={cur_y}
    M400
    RESTORE_GCODE_STATE NAME=STATE_HOME_AXIS

[gcode_macro _HOME_Y]
description: Home axis -- Y with X offset to avoid snags
gcode:
    {% set vars = printer["gcode_macro _MY_VARS"] %}
    SAVE_GCODE_STATE NAME=STATE_HOME_Y
    ; Ensure X is homed before moving to safe X offset
    {% if 'x' not in printer.toolhead.homed_axes %}
        _HOME_AXIS AXIS=X
    {% endif %}
    G90
    ; Move X to safe pos before homing Y
    G0 X{vars.cfg_safe_x_home} F{vars.calc_travel_speed_xy}
    _NOTIFY MSG="System: Homing Y..."
    G28 Y
    ; Move away from Y endstop
    {% set safe_y = printer.toolhead.axis_maximum.y|float - 20.0 %}
    G0 Y{safe_y} F{vars.calc_travel_speed_xy}
    M400
    RESTORE_GCODE_STATE NAME=STATE_HOME_Y

[homing_override]
axes: xyz
gcode:
    {% set vars     = printer["gcode_macro _MY_VARS"] %}
    {% set home_all = 'X' not in params and 'Y' not in params and 'Z' not in params %}
    {% set home_x   = home_all or 'X' in params %}
    {% set home_y   = home_all or 'Y' in params %}
    {% set home_z   = home_all or 'Z' in params %}
    G90
    ; Safe Z-hop only when Z is already homed -- avoids bed crash on first boot
    {% if "z" in printer.toolhead.homed_axes %}
        {% set z_safe = [printer.toolhead.position.z|float + vars.cfg_z_hop_home,
                         printer.toolhead.axis_maximum.z|float]|min %}
        G0 Z{z_safe} F{vars.calc_travel_speed_z}
    {% endif %}
    {% if home_x %}
        _HOME_AXIS AXIS=X
    {% endif %}
    {% if home_y %}
       _HOME_Y ; Home Y axis safely
    {% endif %}
    {% if home_z %}
        ; Guarantee X and Y are homed before probing Z
        {% if (not ("x" in printer.toolhead.homed_axes) and not home_x) or
              (not ("y" in printer.toolhead.homed_axes) and not home_y) %}
            _HOME_AXIS AXIS=X
            _HOME_Y
        {% endif %}
        G0 X{vars.calc_center_x} Y{vars.calc_center_y} F{vars.calc_travel_speed_xy}
        _NOTIFY MSG="System: Homing Z..."
        G28 Z
        G0 Z{vars.cfg_z_hop_home} F{vars.calc_travel_speed_z}
    {% endif %}
    M400
